CREATE OR REPLACE VIEW orders_enriched AS
SELECT
    *,
    YEAR(STR_TO_DATE(order_date, '%Y-%m-%d')) AS order_year,
    MONTH(STR_TO_DATE(order_date, '%Y-%m-%d')) AS order_month,
    DAY(STR_TO_DATE(order_date, '%Y-%m-%d')) AS order_day,
    WEEK(STR_TO_DATE(order_date, '%Y-%m-%d')) AS order_week
FROM ecommerce_orders_raw_table3;
